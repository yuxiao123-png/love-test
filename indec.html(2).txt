    <script>
        let currentQuestion = 0;
        const totalQuestions = document.querySelectorAll('.question').length;
        const progressBar = document.getElementById('progressBar');
        const result = document.getElementById('result');
        
        // 维度分值存储
        const dimensionScores = {
            control: 0,   // 控制欲望
            jealousy: 0,  // 嫉妒强度
            dependence: 0,// 情感依赖
            insecurity: 0 // 关系不安全感
        };
        let totalScore = 0;

        // 更新进度条
        function updateProgress() {
            const progress = ((currentQuestion + 1) / totalQuestions) * 100;
            progressBar.style.width = `${progress}%`;
        }

        // 切换题目
        function switchQuestion(index) {
            document.querySelectorAll('.question').forEach((q, i) => {
                q.classList.toggle('active', i === index);
            });
            currentQuestion = index;
            updateProgress();
        }

        // 下一题
        function nextQuestion() {
            const currentQ = document.querySelector(`.question[data-index="${currentQuestion}"]`);
            const selected = currentQ.querySelector('input[type="radio"]:checked');
            if (!selected) {
                alert('请选择一个答案后继续！');
                return;
            }
            // 累加当前题目对应维度的分值
            const dim = currentQ.dataset.dim;
            dimensionScores[dim] += parseInt(selected.value);
            
            if (currentQuestion < totalQuestions - 1) {
                switchQuestion(currentQuestion + 1);
            }
        }

        // 上一题
        function prevQuestion() {
            if (currentQuestion > 0) {
                // 回退上一题的分值
                const prevQ = document.querySelector(`.question[data-index="${currentQuestion}"]`);
                const selected = prevQ.querySelector('input[type="radio"]:checked');
                if (selected) {
                    const dim = prevQ.dataset.dim;
                    dimensionScores[dim] -= parseInt(selected.value);
                }
                switchQuestion(currentQuestion - 1);
            }
        }

        // 计算结果并展示
        function showResult() {
            const currentQ = document.querySelector(`.question[data-index="${currentQuestion}"]`);
            const selected = currentQ.querySelector('input[type="radio"]:checked');
            if (!selected) {
                alert('请选择一个答案后查看结果！');
                return;
            }
            // 累加最后一题分值
            const dim = currentQ.dataset.dim;
            dimensionScores[dim] += parseInt(selected.value);

            // 计算总分（各维度满分20，总分80）
            totalScore = dimensionScores.control + dimensionScores.jealousy + dimensionScores.dependence + dimensionScores.insecurity;
            
            // 填充维度分值
            document.getElementById('controlScore').innerText = dimensionScores.control;
            document.getElementById('jealousyScore').innerText = dimensionScores.jealousy;
            document.getElementById('dependenceScore').innerText = dimensionScores.dependence;
            document.getElementById('insecurityScore').innerText = dimensionScores.insecurity;
            document.getElementById('totalScore').innerText = totalScore;

            // 判定占有欲等级
            let level = '';
            let desc = '';
            let tip = '';
            if (totalScore <= 16) {
                level = '极低占有欲';
                desc = '你的恋爱占有欲处于<strong>极低</strong>水平，在亲密关系中极度信任伴侣，给予对方充分的自由和空间。你拥有独立的人格和生活，不会因恋爱失去自我，这种状态让伴侣感到轻松。但有时可能会因为过于“佛系”，让伴侣觉得你不够在意这段关系，建议适当表达对彼此的关注。';
                tip = '<strong>相处建议：</strong> 偶尔向伴侣表达关心和思念，让对方感受到你的爱意；在伴侣需要陪伴时，主动投入时间和精力，平衡“自由”与“亲密”的关系。';
            } else if (totalScore <= 32) {
                level = '低占有欲';
                desc = '你的恋爱占有欲处于<strong>低</strong>水平，在亲密关系中能保持理性和信任，尊重伴侣的社交自由和个人空间，同时也有自己的生活和兴趣。这种健康的相处模式让恋爱关系更加轻松和谐，是理想的恋爱状态。';
                tip = '<strong>相处建议：</strong> 继续保持当前的相处模式，在信任对方的同时，也不忘表达自己的情感需求，让彼此的关系在“自由”与“亲密”中找到平衡。';
            } else if (totalScore <= 48) {

                desc = '你的恋爱占有欲处于<strong>适度</strong>水平，在亲密关系中会在意伴侣的行为，偶尔产生小的嫉妒或不安，但能通过自我调节和沟通化解。这种状态既体现了你对这段关系的重视，又不会给伴侣带来过度的压力，是最健康的占有欲水平。';
